  ## glazes
  glazes <- data.frame( 
    "Lichen"  = c(33,0,77,0),
    "Tenoku"  = c(25,80,25,0),
    "Whiting" = c(0,15,10,80))

  ## ingredients
  ingredients <- c("Nepheline.Syenite",
                   "Red.Iron.Oxide",
                   "Flint",
                   "Talc")
  colors <- c("steelblue",
              "red",
              "brown",
              "yellow")
  rownames(glazes) <- ingredients

  ## turn glazes into a matrix
  Glazes <- as.matrix(glazes)
  Glazes # print matrix
  colnames(Glazes)
  rownames(Glazes)
